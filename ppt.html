pigeon-chart syntax (Sample Code)

Rendered Output (Charts)

SELECT [x-axis], [y-axis] FROM [table] 


					<pigeon-chart 
						query="SELECT customerNumber, amount FROM `payments` 
						WHERE customerNumber=103">
					</pigeon-chart>

SELECT [x-axis], [y-axis-series1], [y-axis-series2]... FROM [table] 


					<pigeon-chart 
						query="SELECT productName, buyPrice, MSRP FROM `products` 
						WHERE productLine='Motorcycles'"
						type="area"
						title="Motorcycles"
						subtitle="buyPrice v/s MSRP"
						axis-x-title="Motorcycles"
						axis-y-title="price $"
						show-legend="top">
					</pigeon-chart>

SELECT [x-axis], AGG_FUNC([y-axis]) FROM [table] 


					<pigeon-chart 
						query="SELECT customerNumber, COUNT(orderNumber) FROM `orders`  
						WHERE customerNumber<150 GROUP BY customerNumber"
						type="column"
						title="Orders"
						subtitle="Orders per customer number"
						axis-x-title="customer number"
						axis-y-title="order count"
						show-data-label=true
						show-legend="bottom">
					</pigeon-chart>
